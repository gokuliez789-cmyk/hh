int exists(char arr[][50], int count, char *val) {
    for(int i=0;i<count;i++) {
        if(strcmp(arr[i],val)==0)
            return 1;
    }
    return 0;
}
void addUnique(char arr[][50], int *count, char *val) {
    if(!exists(arr,*count,val)) {
        strcpy(arr[*count],val);
        (*count)++;
    }
}
