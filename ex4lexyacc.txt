%{
#include "y.tab.h"
%}
%%
[0-9]+                 { return ID; }        // numbers
[a-zA-Z][a-zA-Z0-9]*  { return ID; }        // variables
"+"                    { return '+'; }
"*"                    { return '*'; }
"("                    { return '('; }
")"                    { return ')'; }
[ \t]                  ;   // ignore spaces
\n                     return 0;
. { printf("Invalid character: %s\n", yytext); }
%%
int yywrap() {
    return 1;
}
