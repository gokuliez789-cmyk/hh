#include <stdio.h>
#include <string.h>
#include <ctype.h>
#define MAX 20
char g[MAX][MAX], nt[MAX], t[MAX];
char f[26][MAX], l[26][MAX];
int ntc=0, tc=0;
/* ---------- UTIL ---------- */
int has(char a[], char c){ for(int i=0;a[i];i++) if(a[i]==c) return 1; return 0; }
void add(char a[], char c){ if(!has(a,c)){ int n=strlen(a); a[n]=c; a[n+1]='\0'; } }
void addT(char c){ if(!isupper(c)&&c!='-'&&c!='>'&&!has(t,c)) t[tc++]=c; }

/* ---------- FIRSTVT ---------- */

void firstVT(){
    int ch=1;
    while(ch){
        ch=0;
        for(int i=0;i<ntc;i++){
            char A=g[i][0];
            for(int j=3;g[i][j];j++){
                char X=g[i][j];
                if(!isupper(X)){ if(!has(f[A-'A'],X)){ add(f[A-'A'],X); ch=1;} break;}
                for(int k=0;f[X-'A'][k];k++)
                    if(!has(f[A-'A'],f[X-'A'][k])){ add(f[A-'A'],f[X-'A'][k]); ch=1;}
                if(isupper(X)&&g[i][j+1]&&!isupper(g[i][j+1]))
                    add(f[A-'A'],g[i][j+1]);
            }
        }
    }
}

/* ---------- LASTVT ---------- */
void lastVT(){
    int ch=1;
    while(ch){
        ch=0;
        for(int i=0;i<ntc;i++){
            char A=g[i][0]; int len=strlen(g[i]);
            for(int j=len-1;j>=3;j--){
                char X=g[i][j];
                if(!isupper(X)){ if(!has(l[A-'A'],X)){ add(l[A-'A'],X); ch=1;} break;}
                for(int k=0;l[X-'A'][k];k++)
                    if(!has(l[A-'A'],l[X-'A'][k])){ add(l[A-'A'],l[X-'A'][k]); ch=1;}
                if(isupper(X)&&j-1>=3&&!isupper(g[i][j-1]))
                    add(l[A-'A'],g[i][j-1]);
            }
        }
    }
}
